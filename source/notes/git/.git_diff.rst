.. code-block:: sh

    # Working Directory vs Stage
    git diff
    # Working Directory vs Tree
    git diff ${tree}
    # Stage vs Tree
    git diff --cached ${tree:-HEAD}
    # Tree vs Tree
    git diff ${tree1} ${tree2}

    # ... for a file
    git diff ... -- <path>

    # Compare two current folders
    git diff --no-index dir1 dir2
    # can use non-git diff: diff -r dir1 dir2
